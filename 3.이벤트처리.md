# 이벤트 처리

## 엘리먼트에 이벤트핸들러 등록하기
- 인라인 이벤트 처리
    ```html
    <!-- 버튼에서 click 이벤트 발생시 fn1() 함수를 실행시킨다. -->
    <button onclick="fn1();">버튼</button>
    ```

    ```javascript
    function fn1() {
        console.log("버튼이 클릭되었습니다.")
    }
    ```
- 인라인 이벤트 처리
    ```html
    <!-- 버튼에서 click 이벤트 발생시 fn1() 함수를 실행시킨다. -->
    <button onclick="fn1(1);">버튼</button>
    <button onclick="fn1(2);">버튼</button>
    <button onclick="fn1(3);">버튼</button>
    ```

    ```javascript
    function fn1(no) {
        console.log(no);    // 1, 2, 3 중 하나가 출력된다.
    }
    ```
- addEventListener() 메소드로 이벤츠 처리
    ```html
    <button id="btn-1">버튼1</button>
    <button id="btn-2">버튼2</button>
    <button id="btn-3">버튼3</button>
    ```

    ```javascript
    // 엘리먼트를 선택한다.
    let btn1 = document.getElementById("btn-1");
    // 선택된 엘리먼트에서 click 이벤트 발생시 실행될 이벤트핸들러 함수를 등록시킨다.
    btn1.addEventListener('click', function() {
        console.log('버튼1이 클릭되었습니다.');
    });

    // 엘리먼트를 선택한다.
    let btn2 = document.getElementById("btn-2");
    // 선택된 엘리먼트에서 click 이벤트 발생시 실행될 이벤트핸들러 함수를 등록시킨다.
    btn2.addEventListener('click', function() {
        console.log('버튼2이 클릭되었습니다.');
    });

    // 엘리먼트를 선택한다.
    let btn3 = document.getElementById("btn-3");
    // 선택된 엘리먼트에서 click 이벤트 발생시 실행될 이벤트핸들러 함수를 등록시킨다.
    btn3.addEventListener('click', function() {
        console.log('버튼3이 클릭되었습니다.');
    });

    ```
## jquery의 이벤트 처리
- jquery의 이벤트 처리 메소드
    ```javascript
    // 선택자로 선택된 모든 엘리먼트에서 지정된 이벤트가 발생할 때마다 매번 실행될 이벤트핸들러 함수를 등록시킨다.
    $("선택자").on("이벤트이름", 이벤트핸들러함수);

    // 선택자1의 내부에 이미 존재하거나 미래에 추가될 엘리먼트 중에서 선택자2에 해당 엘리먼트에서 지정한 이벤트가 발생할 때마다 실행될 이벤트핸들러 함수를 등록시킨다.
    // 선택자1에 해당하는 엘리먼트는 이미 존재하고 있어야 한다.
    // 선택자2에 해당하는 엘리먼트는 이미 존재할 수도 있고, 미래에 추가될 수도 있다.
    $("선택자1").on("이벤트이름", "선택자2", 이벤트핸들러함수);

    // 선택자로 선택된 모든 엘리먼트에서 지정한 이벤트가 발생핬을 때 딱 한번 실행될 이벤트핸들러 함수를 등록시킨다.
    $("선택자").one("이벤트이름", 이벤트핸들러함수);

    // 선택자로 선택된 모든 엘리먼트에서 지정한 이벤트와 관련된 이벤트핸들러 함수를 제거한다.
    $("선택자").off("이벤트이름");

    // 선택자로 선택된 모든 엘리먼트에서 지정한 이벤트를 소프트웨어적인 방법으로 강제로 발생시킨다.
    $("선택자").trigger("이벤트이름");

    // 이벤트 처리 편리메소드를 이용해서 선택자로 선택된 모든 엘리먼트에서 해당 이벤트가 발생될 때마다 실행되는 이벤트핸들러 함수를 등록시킨다.
    $("선택자").click(이벤트핸들러함수);
    $("선택자").change(이벤트핸들러함수);
    $("선택자").submit(이벤트핸들러함수);
    $("선택자").mouseenter(이벤트핸들러함수);
    $("선택자").mouseleave(이벤트핸들러함수);
    $("선택자").keyup(이벤트핸들러함수);
    ```
- jquery의 이벤트핸들러 함수
```javascript
// 이벤트핸들러 함수는 매개변수 없이 작성할 수 있다.
$("선택자").on("이벤트이름", function() {

});

// 이벤트핸들러 함수는 매개변수로 event 객체를 전달받도록 작성할 수 있다.
$("선택자").on("이벤트이름", function(event) {
   
});

// 이벤트핸들러 함수 내부에서 사용하는 this는 언제나 이벤트가 발생한 그 엘리먼트 객체다.
$("선택자").on("이벤트이름", function() {
    console.log(this);  // 엘리먼트객체(이벤트가 발생한 그 엘리먼트객체다.)
});

// 이벤트핸들러 함수는 true 혹은 false값을 반환값으로 사용할 수 있다.
$("선택자").on("이벤트이름", function() {
    // false 값이 반환되면
    // 1. 이벤트가 발생한 엘리먼트에서 이벤트와 관련된 기본동작이 실행되지 않는다.
    //      <a> 엘리먼트인 경우 click 이벤트가 발생하더라도 링크로 이동하지 않는다.
    //      <form> 엘리먼트인 경우 submit 이벤트가 발생하더라고 폼이 제출되지 않는다.
    // 2. 이벤트가 부모 엘리먼트로 전파되지 않는다.(이벤트버블링이 방지된다.)
    return false;
});
```
